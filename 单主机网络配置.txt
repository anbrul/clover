单主机上配置network让容器互通
docker network create dev_bridge
容器都使用这个bridge，例如：
version: '3'
services:
  redis:
    image: redis:alpine
    restart: always
    volumes:
      - ./data:/data
    ports:
      - 6379:6379
    command: redis-server --appendonly no --maxmemory-policy allkeys-lru --requirepass "redispassword"
    networks:
      - "dev_bridge"
networks:
  dev_bridge:
    external: true
